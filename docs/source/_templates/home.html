import numpy as np
from fury.actor import sphere
from fury.window import show

rng = np.random.default_rng()
centers = rng.uniform(-50, 50, size=(1000, 3))
colors = rng.uniform(0, 1, size=(1000, 3))

actor = sphere(centers, colors=colors, radii=1.5)
show(actors=[actor])
